Huepicker

A hassle-free colorpicker control, ~8.1KB gzipped.

Example

Install

Download

or

npm install huepicker

Use

html

<!--be sure to use a div rather than an input!-->
<div id="huepicker"></div>

js

import huepicker from 'huepicker'

huepicker({
  el: document.getElementById('huepicker'),
  value: 'red',
  onChange: function(color) {
    console.log('changed', color)
  }
})
